int pickingNumbers(int a_count, int* a) {
   int i,j,c=0,max,b[a_count];
   for(i=0;i<a_count;i++)
   {
       for(j=0;j<a_count;j++)
       {
           if(a[i]==a[j] || (a[i]+1)==a[j])
            c++;
         }
         b[i]=c;
         c=0;
    }   
   max=b[0];
   for(i=1;i<a_count;i++)
   {
       if(max<b[i])
        max=b[i];
   }
   return max;
}
