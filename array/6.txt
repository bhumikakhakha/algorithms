int birthdayCakeCandles(int candles_count, int* candles) {
    int i,max=candles[0],count=0;
    for(i=0;i<candles_count;i++)
    {
        if(max<candles[i])
           max=candles[i];
    }
    for(i=0;i<candles_count;i++)
    {
        if(max==candles[i])
         count++;
    }
    return count;

}
